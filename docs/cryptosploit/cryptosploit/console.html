 81    def do_shell(self, arg):
 82        """
 83        Any shell command.
 84        Example: ls -la
 85        """
 86        proc = Popen(
 87            arg,
 88            stderr=PIPE,
 89            shell=True,
 90            stdin=PIPE,
 91            stdout=PIPE,
 92            universal_newlines=True,
 93            text=True,
 94        )
 95        Printer.exec(f"Executing '{arg}'")
 96        for line in iter(proc.stdout.readline, ""):
 97            print(line, end="")
 98        return_code = proc.wait()
 99        if return_code == 127:
100            raise UnknownCommandError("Unknown command")
101        else:
102            for line in iter(proc.stderr.readline, ""):
103                print(line, "\n")
104            proc.stderr.close()
105        return False

107    def do_use(self, module_path: str):
108        """
109        Load module
110        Example: use symmetric.rot
111        """
112        try:
113            self.module = import_module("cryptosploit_modules." + module_path).module
114            venv_path = os.path.join(os.path.dirname(getfile(self.module.__class__)), "venv")
115            for root, dirs, _ in os.walk(venv_path):
116                if "site-packages" in dirs:
117                    path[0] = os.path.join(root, "site-packages")
118                    break
119            else:
120                path[0] = ""
121            self.variables = self.module.env
122            self.prompt = f"crsconsole {colorize_strings(f'({module_path})', fg=SGR.COLOR.FOREGROUND.PURPLE)}> "
123            Printer.info("Module loaded successfully")
124        except ModuleNotFoundError:
125            raise ModuleError("No such module")
126        except AttributeError:
127            raise ModuleError("Not a module")
128        return False

130    def do_search(self, name):
131        """
132        Search modules by keyword.
133        Example: search rot
134        """
135        pattern = f".*{name}.*"
136        try:
137            r = compile(pattern)
138            found = list(filter(r.match, self.modules_list))
139            Printer.info("\n".join(found) if found else f"No results for {name}")
140            return False
141        except error:
142            raise ArgError("Invalid regex")

160    def do_set(self, arg):
161        """
162        Set the value of a variable.
163        Example: set ciphertext OwO
164        """
165        arg = arg.split(maxsplit=1)
166        if len(arg) == 2:
167            name, value = arg
168            if self.variables:
169                if name in self.variables:
170                    self.variables.set_var(name, value)
171                    Printer.info(f"Setting {name} -> {value}")
172                    return False
173                else:
174                    raise ArgError("No such variable")
175            raise ModuleError("Module is not loaded")
176        raise ArgError("Value is not set")

186                Printer.info(f"Setting {name} -> None")
187                return False

189                raise ArgError("No such variable")

221    @staticmethod
222    def explore_paths(line, only_dirs):
223        text = (line.split(" "))[-1]
224        if text in ("..", "."):
225            return (
226                [os.path.join(".", ""), os.path.join("..", "")]
227                if text == "."
228                else [os.path.join("..", "")]
229            )
230        if only_dirs:
231            paths = filter(
232                lambda x: os.path.isdir(os.path.join(os.path.dirname(text), x)),
233                listdir(os.path.dirname(text) or "."),
234            )
235        else:
236            paths = listdir(os.path.dirname(text) or ".")
237        founded = list(
238            map(
239                lambda a: os.path.join(a, "") if os.path.isdir(a) else a,
240                filter(lambda x: x.startswith(os.path.split(text)[-1]), os.paths),
241            )
242        )
243        return founded

257    def complete_set(self, text, line, begidx, endidx):
258        if " " not in line[:begidx].strip():
259            founded = []
260            if self.variables:
261                for varname in iter(self.variables):
262                    if varname.startswith(text):
263                        founded.append(varname)
264                return founded
265        else:
266            return self.explore_paths(line, False)

269        return self.complete_set(text, line, begidx, endidx)
